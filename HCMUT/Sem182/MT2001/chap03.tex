\chapter{Discrete Random Variables and Probability Distributions}

\hi{Probability Distributions and Probability Mass Functions}
  \hii{Probability Mass Function}
    \par For a discrete random variable $X$ with possible values $x_1, x_2,
      \ldots, x_n$, a \tb{probability mass function} is a function such that
    \begin{enumerate}[(1)]
      \item $f(x_i) \geq 0$
      \item $\SUM{_{i = 1}^{n}} f(x_i) = 1$
      \item $f(x_i) = P(X = x_i)$
    \end{enumerate}

\hi{Cumulative Distribution Functions}
  \par Theh \tb{cummulative distribution fuction} of a discrete random variable
    $X$, denoted as $F(x)$, is
  \begin{align*}
    F(x) = P(X \leq x) = \SUM{_{x_i < x}} f(x_i)
  \end{align*}
  \par For a discrete random variable $X$, $F(x)$ satisfies the following
    properties:
  \begin{enumerate}[(1)]
    \item $F(x) = P(X \leq x) = \SUM{_{x_i < x}} f(x_i)$
    \item $0 \leq F(x) \leq 1$
    \item If $x \leq y$, then $F(x) \leq F(y)$
  \end{enumerate}
