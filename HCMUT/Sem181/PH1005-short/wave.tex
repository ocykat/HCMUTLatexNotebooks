\chapter{Mechanical Wave}


\begin{multicols}{3}


\hi{Wave Function}
  \hii{Wave function}
     \begin{align*}
        y(x, t) &= A\cos\bigg[\omega\bigg(t - \frac{x}{v}\bigg)\bigg] \\
                &= A\cos\bigg[2\pi\bigg(\frac{x}{\lambda} - \frac{t}{T}\bigg)\bigg]
                &= A\cos(\omega t - kx)
     \end{align*}
  \hii{Wave number}
    \begin{align*}
      k = \frac{2\pi}{\lambda}
    \end{align*}
    \begin{align*}
      \omega = vk
    \end{align*}


\hi{Speed of Traverse Wave}
  \begin{align*}
    v = \sqrt{\frac{F}{\mu}}
  \end{align*}
  \begin{itemize}
    \item $v$: speed of traverse wave $[m/s]$
    \item $F$: force $[N]$
    \item $\mu$: mass per unit length $[kg/m]$
  \end{itemize}


\hi{Power}
  \begin{align*}
    \mathcal{\avg{P}} = \frac{1}{2} \mu v \omega^2 A^2
  \end{align*}

\end{multicols}
