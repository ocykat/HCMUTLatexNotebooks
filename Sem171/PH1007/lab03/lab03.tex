\hi{Data Analysis \& Results}
    \hii{Measurements}

    \hiiBEGIN{Calculations}
        \hiii{Calculate the values}
            \begin{align*}
                \avg{f_{fric.}} & = \avg{m} \cdot \avg{g} \cdot \dfrac{\avg{h_{1}} - \avg{h_{2}}}{\avg{h_{1}} + \avg{h_{2}}} \\
                & = 0.2385 \mul 9.81 \mul \dfrac{0.674 - 0.583}{0.674 + 0.583} \\
                & = 0.17 N 
            \end{align*}
            \begin{align*}
                \avg{I} & = \dfrac{\avg{m} \cdot \avg{d}^{2}}{4} \cdot \bigg[ \avg{g} \cdot \avg{t}^{2} \cdot \dfrac{\avg{h_{2}}}{\avg{h_{1}} \cdot (\avg{h_{1}} + \avg{h_{2}})} - 1 \bigg] \\
                & = \dfrac{0.2385 \times (8.07 \times 10^{-3})^{2}}{4} \times \bigg[ 9.81 \times 6.725^{2} \times \dfrac{0.583}{0.674 \times (0.674 + 0.583)} - 1 \bigg] \\
                & = 1.182 \times 10^{-3} (kg \cdot m^{2})
            \end{align*}

        \hiiiBEGIN{Calculate the errors}
            \hiv{Total errors}
                \begin{align*}
                    \Dt m     &= (\Dt m)_{instrument}     + \avg{\Dt m}     &= 0.0001 + 0    &= 0.0001 (kg) & \\
                    \Dt h_{1} &= (\Dt h_{1})_{instrument} + \avg{\Dt h_{1}} &= 0.001 + 0     &= 0.001 (m) & \\
                    \Dt h_{2} &= (\Dt h_{2})_{instrument} + \avg{\Dt h_{2}} &= 0.001 + 0.003 &= 0.004 (m) & \\
                    \Dt t     &= (\Dt t)_{instrument}     + \avg{\Dt t}     &= 0.001 + 0.106 &= 0.107 (s) & \\
                    \Dt d     &= (\Dt d)_{instrument}     + \avg{\Dt d}     &= 0.02 + 0.08   &= 0.1 (10^{-3} m) & \\
                \end{align*}

            \hiv{Relative errors}
                \begin{align*}
                    \dfrac{\Dt f_{fric.}}{\avg{f_{fric.}}} 
                    & = \dfrac{\Dt m}{\avg{m}} + \dfrac{\Dt g}{\avg{g}} +
                        \bigg| \dfrac{1}{\avg{h_{1}} - \avg{h_{2}}} - \dfrac{1}{\avg{h_{1}} + \avg{h_{2}}} \bigg| \cdot \Dt h_{1}
                        \bigg| \dfrac{-1}{\avg{h_{1}} - \avg{h_{2}}} - \dfrac{1}{\avg{h_{1}} + \avg{h_{2}}} \bigg| \cdot \Dt h_{2} \\
                    & = \dfrac{0.0001}{0.2385} + \dfrac{0.01}{2 \times 9.81} +
                        \bigg| \dfrac{1}{0.674 - 0.583} - \dfrac{1}{0.674 + 0.583} \bigg| \times 0.001 \\
                    & + \bigg| \dfrac{-1}{0.674 - 0.583} - \dfrac{1}{0.674 + 0.583} \bigg| \times 0.004 \\
                    & = 2.29 \% \\
                    \Rightarrow & \Dt f_{fric.} = \avg{f_{fric.}} \times 2.29 \% = 0.17 \times 2.29 \% = 0.004 (N)
                \end{align*}
                \begin{align*}
                    \relerr{I} & = \relerr{m} + 2 \cdot \relerr{d} + \relerr{g} + 2 \cdot \relerr{t} +
                        \bigg| \dfrac{-1}{\avg{h_{1}}} - \dfrac{1}{\avg{h_{1}} + \avg{h_{2}}} \bigg| \cdot \Dt h_{1} +
                        \bigg| \dfrac{1}{\avg{h_{2}}} - \dfrac{1}{\avg{h_{1}} + \avg{h_{2}}} \bigg| \cdot \Dt h_{2} \\
                    & = \dfrac{0.0001}{0.2385} + 2 \times \dfrac{0.1 \times 10^{-3}}{8.07} + \dfrac{0.01}{2 \times 9.81} + 2 \times \dfrac{0.107}{6.725} \\
                    & + \bigg| \dfrac{-1}{0.674} - \dfrac{1}{0.674 + 0.583} \bigg| \times \Dt 0.001 
                      + \bigg| \dfrac{1}{0.583} - \dfrac{1}{0.674 + 0.583} \bigg| \cdot \Dt 0.004 \\
                    & = 3.87 \% \\
                    \Rightarrow & \Dt I = \avg{I} \times 3.87 \% = 1.182 \times 10^{-3} \times 3.87 \% = 0.046 \times 10^{-3} (kg \cdot m^{2})
                \end{align*}
        \hiiiEND

    \hiiEND

    \hii{Results}

            f_{fric.} = \avg{f_{fric.}} \pm \Dt f_{fric.} = 0.17 \pm 0.004 (N) \\
        \end{align*}
        \begin{align*}
            I = \avg{I} \pm \Dt I = 1.182 \pm 0.046 \quad (10^{-3} kg \cdot m^{2})
        \end{align*}
