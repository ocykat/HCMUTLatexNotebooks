\hi{Objective}
    \par Determine the mass density of some symmetrical objects using a vernier
    calliper and a balanced scale.

\hi{Data Analysis \& Results}
    \hiiBEGIN{Determine the mass density of object 1}
        \hiii{Determine the volume of object 1}
            \begin{center}
                \begin{tabular}{|c|c|c|c|c|c|c|}
                    \hline
                    \multicolumn{7}{|c|}{Precision of calliper: 0.02 mm} \\
                    \hline
                    \multirow{2}{*}{Meas. No.} & $D$ & $\Dt D$ & $d$ & $\Dt d$ & $h$ & $\Dt h$ \\
                    & $(10^{-3} m)$ & $(10^{-3} m)$ & $(10^{-3} m)$ & $(10^{-3} m)$ & $(10^{-3} m)$ & $(10^{-3} m)$ \\
                    \hline
                    1       & 33.02 & 0.05 & 20.70 & 0.01 & 29.52 & 0.02 \\
                    \hline
                    2       & 32.98 & 0.01 & 20.78 & 0.07 & 29.56 & 0.02 \\
                    \hline
                    3       & 32.90 & 0.07 & 20.66 & 0.05 & 29.54 & 0.00 \\
                    \hline
                    Average & 32.97 & 0.04 & 20.71 & 0.04 & 29.54 & 0.01 \\
                    \hline
                \end{tabular}
            \end{center}

            \begin{itemize}
                \item Calculate the total errors of measurement of the outer radius $D$,
                    the inner radius $d$, and the height $h$:
                    {\setlength{\abovedisplayskip}{0pt}
                    \begin{align*}
                        \Dt D & = (\Dt D)_{instrument} + \avg{\Dt D} & = 0.02 + 0.04 & = 0.06 & (10^{-3} m) \\
                        \Dt d & = (\Dt d)_{instrument} + \avg{\Dt d} & = 0.02 + 0.04 & = 0.06 & (10^{-3} m) \\
                        \Dt h & = (\Dt h)_{instrument} + \avg{\Dt h} & = 0.02 + 0.01 & = 0.03 & (10^{-3} m) 
                    \end{align*}}
                \item Calculate the relative error and the object's volume:
                    {\setlength{\abovedisplayskip}{0pt}
                    \begin{flalign*}
                        \delta = \dfrac{\Dt V}{\avg{V}}
                        & = \dfrac{\Dt \pi}{\pi} + 2 \mul \dfrac{\avg{D} \Dt D + \avg{d} \Dt d}{\avg{D^{2}} - \avg{d^{2}}} + \dfrac{\Dt h}{\avg{h}} \\
                        & = \dfrac{0.01}{2\pi} + 2 \mul \dfrac{32.97 \mul 0.06 + 20.71 \mul 0.06}{32.97^{2} - 20.71^{2}} + \dfrac{0.03}{29.54} \\
                        & = 0.012 \\
                    \end{flalign*}
                    \begin{flalign*}
                        \avg{V} & = \dfrac{\pi}{4} (\avg{D^{2}} - \avg{d^{2}}) \mul \avg{h} \\
                        & = \dfrac{\pi}{4} (32.97^{2} - 20.71^{2}) \mul 29.54 \\
                        & = 15268 \qquad (10^{-9} m^{3}) \\
                    \end{flalign*}
                    \begin{flalign*}
                        \Dt V & = \delta \mul \avg{V} = 0.012 \mul 15268 = 183 \qquad (10^{-9} m^{3}) \\
                    \end{flalign*}}
                \item Write the result of the ring volume measurement:
                    \begin{flalign*}
                        V = \avg{V} \pm \Dt V = 15268 \pm 183 \qquad (10^{-9} m^{3}) \\
                    \end{flalign*}
           \end{itemize}

        \hiii{Determine the mass of object 1}
            \begin{center}
                \begin{tabular}{|c|c|c|}
                    \hline
                    \multirow{2}{*}{Meas. No.} & $m$ & $\Dt m$ \\
                    & $(10^{-3} kg)$ & $(10^{-3} kg)$ \\
                    \hline
                    1       & 151.3 & 0.2 \\
                    \hline
                    2       & 151.1 & 0.0 \\
                    \hline
                    3       & 151.0 & 0.1 \\
                    \hline
                    Average & 151.1 & 0.1 \\
                    \hline
               \end{tabular}
            \end{center}

            \begin{itemize}
                \item Calculate total errors:
                    \begin{flalign*}
                        \Dt m = (\Dt m)_{instrument} + \avg{\Dt m} = 0.1 + 0.1 = 0.2 \qquad (10^{-3} kg)
                    \end{flalign*}
                \item Wirte the result of the mass measurement
                    \begin{flalign*}
                        m = \avg{m} \pm \Dt m = 151.1 \pm 0.2 \qquad (10^{-3} kg)
                    \end{flalign*}
            \end{itemize}

        \hiii{Determine the mass density of object 1}
            \begin{flalign*}
                \avg{\rho} & = \dfrac{\avg{m}}{\avg{V}} \\
                & = \dfrac{151.1 \mul 10^{-3}}{15268 \mul 10^{-9}} = 9897 \qquad (kg/m^{3}) \\
            \end{flalign*}
            \begin{flalign*}
                \dfrac{\Dt \rho}{\avg{\rho}} = \dfrac{\Dt m}{\avg{m}} + \dfrac{\Dt V}{\avg{V}}
                = \dfrac{0.2}{151.1} + \dfrac{183}{15268} = 0.013 \\
                \ra \Dt \rho = 0.013 \mul \avg{\rho} = 0.013 \mul 9897 = 129 \qquad (kg/m^{3}) \\
            \end{flalign*}
            \begin{flalign*}
                \rho = \avg{\rho} \pm \Dt \rho = 9.897 \pm 0.129 \qquad (10^{3} kg/m^{3}) \\
            \end{flalign*}
