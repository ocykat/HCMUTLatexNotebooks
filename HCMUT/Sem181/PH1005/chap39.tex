\chapter{Particles Behaving As Waves}

\hi{Electron Waves}
  \par de Broglie's reasoning: Electrons and protons, which we usually think
    as \tb{particles}, may in some situation behave like \tb{waves}.
  \par The \tb{de Broglie wavelength of a particle}:
  \begin{equation}
    \lambda = \frac{h}{p} = \frac{h}{mv}
  \end{equation}
  where
  \begin{itemize}
    \item $h$: Planck's constant
  \end{itemize}
  \par The energy of the particle, like photons, related to its frequency:
  \begin{equation}
    E = hf
  \end{equation}

\hi{Energy Levels and the Bohr Model of the Atom}
  \begin{equation}
    hf = \frac{hc}{\lambda} = E_{i} - E_{f}
  \end{equation}
