\chapter{Electric Charge and Electric Field}


\hi{Electric Charge}
  \hii{Interaction}
    \par Charges of the same sign repel each other. Charges of different sign
      attract each other.

  \hii{Proton, neutron, electron}

  \hii{Conservation of Electric Charge}
    \par The algebraic sum of all the electric charges in any closed system
      is constant.

\hi{Conductors, Insulators, and Induced Charges}


\hi{Coulomb's Law}
  \begin{eqbox}
    F = \frac{1}{4 \pi \epsilon_0} \cdot \frac{q_1 q_2}{r^2}
      = k \frac{q_1 q_2}{r^2}
  \end{eqbox}


\hi{Electric Field and Electric Forces}
  \par The electric force on a charged body is exerted by the electric field
    created by \ti{other} charged bodies.
  \par Electric Field:
  \begin{eqbox}
    \vec{E} = \frac{\vec{F_0}}{q_0}
      = \frac{1}{4 \pi \epsilon_0} \frac{q}{r^2} \hat{r}
  \end{eqbox}
