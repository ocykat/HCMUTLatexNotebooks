\chapter{Magnetic Field}


\hi{Magnetic Field}
  \hii{Definition}
    \par A moving charge or a current would create a \tb{magnetic field} in the
      surrounding space.
    \par A magnetic field exerts a force on any moving charge or current that is
      present in the field.
  \hii{Magnetic Force on Moving Charge}
    \par Vector form
    \begin{eqbox}
      \vec{F} = q\vec{v}\vec{B}
    \end{eqbox}
    \par Magnitude
    \begin{eqbox}
      F = \abs{q}v_{\perp} B = qvB \sin(\vec{v}, \vec{B})
    \end{eqbox}
    \begin{itemize}
      \item $B$: $[T]$ (Tesla)
    \end{itemize}


\hi{Magnetic Field Lines and Magnetic Flux}
  \hii{Magnetic Field Lines}
    \par \tb{Magnetic field lines} are drawn so that the line through any point
    is tangent to the magnetic field vector $\vec{B}$ at that point.
      \begin{eqbox}
        \Phi_B = \INT_{}^{} \vec{B} d\vec{A}
      \end{eqbox}

\hi{Magnetic Force on a Current-Carrying Conductor}
    \par Vector form
      \begin{eqbox}
        \vec{F} = I\vec{l}\vec{B}
      \end{eqbox}
    \par Magnitude
      \begin{eqbox}
        F = \abs{I}lB_{\perp} = IlB \sin(\vec{l}, \vec{B})
      \end{eqbox}
