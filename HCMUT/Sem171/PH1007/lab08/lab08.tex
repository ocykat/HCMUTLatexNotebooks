\hi{Objective}
    \par Using the microscope to measure the refractive index of a thin film.

\hi{Measurement Data}
    \begin{itemize}
        \item Precision of the panme ruler: $0.01 (mm)$
        \item Precision of the ring ruler of the microscope: $0.001 (mm)$
    \end{itemize}
    \begin{center}
        \begin{tabular}{|c|c|c|c|c|c|c|c|}
            \hline
            \mRow{2}{*}{No.} & \mCol{5}{c|}{Effective thickness $(mm)$} & \mCol{2}{c|}{Real thickness $(mm)$} \\
            \hline
            & \quad $N$ \quad & \quad $l_{0}$ \quad & \quad $l$ \quad & \quad $d_{1}$ \quad & \quad $\Dt d_{1}$ \quad & \quad $d$ \quad & \quad $\Dt d$ \quad \\

\hi{Calculation}
    \hii{Value}
        \begin{align*}
            \avg{n} = \frac{\avg{d}}{\avg{d_{1}}} = \frac{1.93}{1.236} = 1.561
        \end{align*}

    \hii{Deviation}
        \begin{itemize}
            \item Measurement error of $d$: \\
                \begin{align*}
                    & \Dt d = (\Dt d)_{instrument} + \avg{\Dt d} = 0.01 + 0.01 = 0.02 (mm)
                \end{align*}
            \item Measurement error of $d_{1}$: \\
                \begin{align*}
                    & \Dt d_{1} = (\Dt d_{1})_{instrument} + \avg{\Dt d_{1}} = 0.001 + 0.008 = 0.009 (mm)
                \end{align*}
        \end{itemize}
        \begin{align*}
            \relerr{n} = \relerr{d} + \relerr{d_{1}} = \frac{0.02}{1.93} + \frac{0.009}{1.236} = 1.76 \% \\
            \ra \Dt n = \avg{n} \cdot 1.76 \% = 0.027
        \end{align*}

\hi{Result}
    \begin{tcolorbox}
        \begin{align*}
            n = \avg{n} \pm \Dt n = 1.561 + 0.027
        \end{align*}
    \end{tcolorbox}
