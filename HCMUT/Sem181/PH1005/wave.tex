\chapter{Wave Formulas}

\hi{Wave Function}
  \hii{Wave function}
     \begin{align*}
        y(x, t) &= A\cos\bigg[\omega\bigg(t - \frac{x}{v}\bigg)\bigg] \\
                &= A\cos\bigg[2\pi\bigg(\frac{x}{\lambda} - \frac{t}{T}\bigg)\bigg]
                &= A\cos(\omega t - kx)
     \end{align*}
  \hii{Wave number}
    \begin{align*}
      k = \frac{2\pi}{\lambda}
    \end{align*}
    \begin{align*}
      \omega = vk
    \end{align*}
